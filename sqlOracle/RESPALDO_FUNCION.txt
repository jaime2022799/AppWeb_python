create or replace FUNCTION VALIDACION_SIGNUP (V_EMAIL  VARCHAR)
  
 RETURN VARCHAR
IS

 --V_OK VARCHAR2(10);
-- V_NO_OK VARCHAR2(10);
 V_VALIDACION VARCHAR(100);



BEGIN 

  SELECT EMAIL INTO V_VALIDACION
  FROM SIGNUP
  WHERE EMAIL=V_EMAIL;
  RETURN V_VALIDACION;
  
  --V_EMAIL := SIGNUP.EMAIL;
  
  IF V_EMAIL = '' 
  THEN 
   RETURN 'OK';
  ELSE
   RETURN 'NO OK';
  END IF;



END;
